$(document).ready(function() {
	var search = { 
		listPane: null,
		inputForm: null,
		closeBttn: null,
		lastCall: null,
		selectSite: 0,
		init: function() {
			var prevText = '';
			this.inputForm = $('div.h_searchBar').find('input[type="search"]');
			this.searchBtn = $('div.h_searchBar').find('button.submit');
			this.listPane = $('div.h_searchBar').find('div.search_result');
			this.closeBttn = $('div.h_searchBar').find('div.command > a.close');
			this.lastCall = new Date();
			this.inputForm.on({
				keyup: function(e) {
					//검색 
					var str = $.trim($(this).val());
					if (prevText != str) {
						prevText = str;
						if (str.length) {
							search.doSearch(str, ['site']);
						} else {
							search.listPane.removeClass('list show').find('ul').html('');//초기화  
						}
						this.selectSite = 0; 
						$('.search_result ul').scrollTop(0);
					}
					
					// ↑: 38 ↓ : 40
					if(e.keyCode == 40 || e.keyCode == 38){
						searchResult = $('div.search_result').find('ul > li');  
						searchCount = searchResult.length;
						searchResult.removeClass('active');
						if(e.keyCode == 40){ 
							if(searchCount > this.selectSite){
								searchResult.eq(this.selectSite).addClass('active');
								this.selectSite++;  
							}else if(searchCount == this.selectSite){
								searchResult.eq(this.selectSite-1).addClass('active');
							}
						}else{
							if(this.selectSite == 0 || this.selectSite == 1){
								if(this.selectSite){
									searchResult.eq(0).addClass('active');
								}
								
								scrollpx = 0;  
							}else{
								this.selectSite--;
								searchResult.eq(this.selectSite-1).addClass('active'); 
							}
						}  

						//console.log('t:'+searchCount+'//c:'+this.selectSite); 

						scrollpx = Math.floor(this.selectSite)*30; 
						if(scrollpx > 0){
							$('.search_result ul').scrollTop(scrollpx-90); 
						}
					}else if(e.keyCode == 13){// enter 
						if(this.selectSite > 0){
							//onsole.log(searchResult.eq(this.selectSite-1).find('a').attr('href'));
							location.href = searchResult.eq(this.selectSite-1).find('a').attr('href');
						}
					}else{	//초기화 
						this.selectSite = 0; 
						$('.search_result ul').scrollTop(0);
					}


				},
				focus: function() { 
					prevText = '';
					//$(this).keyup(); 
				},
				blur: function() {   
					//익스 예외처리 
					// if(jQuery.browser.msie){ 
						if(search.listPane.hasClass('nolist')){
							search.listPane.removeClass('list show nolist');
							//초기화 
							this.selectSite = 0; 
							$('.search_result ul').scrollTop(0);
						}
					//}
				} 
			});
			this.searchBtn.on('click', function(e) {
				e.preventDefault();
				search.inputForm.trigger('keyup');
			});
			this.listPane.on('mousedown mouseup click', 'a', function(e) { 
				//console.log(e);
				search.inputForm.focus();
				if (e.type != 'click') {
					e.preventDefault();
				}

			});
			this.closeBttn.on('click', function(e) { 
				search.listPane.removeClass('show');
				e.preventDefault();
			});
			var searchWord = '';
			$('#h_searchBar_form').submit(function(e) { 
				return false;
			});
		},
		doSearch: function(str, option) {
			this.lastCall = new Date();
			var ajaxUrl = $('body').data('site') == 'pick' ? 'https://pick.inven.co.kr/quicklink/search' : 'http://www.inven.co.kr/common/quicklink/search.ajax.php';
			$.ajax({
				type: 'POST',
				url: ajaxUrl,
				dataType: 'json',
				data: { search: str, target: option, searchID: this.lastCall, maxLength: 20 }
			}).success(this.onSuccess).fail(this.onFail);
		},
		onSuccess: function(r) {
			var part, list, i, txt, data = r.data, ul, nolist = true;
			ul = search.listPane.removeClass('list show').find('ul');
			ul.html(''); 
			if (r) {
				if (r.data && r.searchID) {
					if (r.searchID != search.lastCall) { return 0; }
					for (part in data) {
						if (!data.hasOwnProperty(part)) { continue; }
						list = data[part];
						if (list.length || typeof list.constructor == 'function') {
							for (i in list) {
								if (list.hasOwnProperty(i)) {
									if (nolist) { nolist = false; }
									txt = list[i];

(function ($) {
    var i = 0;
    var len = $(".panel>li").length;
    var isAnimated = false;
    var timer;

    timer = setInterval(function(){
        if(i==len-1){
            i=0;
        }else{
            i++;
        }
        doNext(i);
    },5000);

    $(".next").on("click",function(e){
        e.preventDefault();
        i = $(".panel>li.on").index();

        if(i==len-1){
            i=0;
        }else{
            i++;
        }
        stop();
        doNext(i);
    });

    $(".prev").on("click",function(e){
        e.preventDefault();
        i = $(".panel>li.on").index();

        if(i==-len+1){
            i=0;
        }else{
            i--;
        }
        stop();
        doPrev(i);
    });


    function doNext(i){
        if(!isAnimated){
            isAnimated=true;
            nextSlide(i);
        }
    }

    function doPrev(i){
        if(!isAnimated){
            isAnimated=true;
            prevSlide(i);
        }
    }
